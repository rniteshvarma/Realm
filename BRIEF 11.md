FIX A — DEVICE TIER DETECTION SYSTEM  
──────────────────────────────────────────────

Create a single utility that runs once at startup  
and returns a performance tier. ALL visual systems  
read from this. Nothing overrides it.

\`\`\`javascript  
  // /src/utils/deviceTier.js

  export function getDeviceTier() {  
    const gl \= document.createElement('canvas')  
      .getContext('webgl')  
    const debugInfo \= gl?.getExtension(  
      'WEBGL\_debug\_renderer\_info'  
    )  
    const gpu \= debugInfo  
      ? gl.getParameter(  
          debugInfo.UNMASKED\_RENDERER\_WEBGL  
        ).toLowerCase()  
      : ''

    const isMobile \= /android|iphone|ipad|ipod/i  
      .test(navigator.userAgent)  
    const isLowEndGPU \= /mali|adreno \[23\]/i.test(gpu)  
    const cores \= navigator.hardwareConcurrency || 4  
    const memory \= navigator.deviceMemory || 4

    // Tier 3: High-end desktop (M4 Pro, RTX etc)  
    if (\!isMobile && cores \>= 8 && memory \>= 8\)  
      return 3

    // Tier 2: Mid-range (modern laptop, recent phone)  
    if (\!isLowEndGPU && cores \>= 4 && memory \>= 4\)  
      return 2

    // Tier 1: Low-end (older phones, budget laptops)  
    return 1  
  }

  export const TIER \= getDeviceTier()

  // Settings matrix — every system reads from this  
  export const SETTINGS \= {  
    // Particle counts  
    nebulaParticles:   \[30000, 70000, 120000\]\[TIER-1\],  
    forgeParticles:    \[512,   1024,  3000  \]\[TIER-1\],  
    starfieldCount:    \[500,   1500,  3000  \]\[TIER-1\],

    // Pixel ratio cap  
    pixelRatio: Math.min(  
      window.devicePixelRatio,  
      \[1.0, 1.5, 2.0\]\[TIER-1\]  
    ),

    // Geometry quality  
    sphereSegments:    \[16, 32, 64\]\[TIER-1\],  
    tunnelSegments:    \[32, 48, 80\]\[TIER-1\],

    // Post-processing  
    bloomEnabled:      \[false, true,  true \]\[TIER-1\],  
    bloomIntensity:    \[0,     0.8,   1.4  \]\[TIER-1\],  
    chromaticAberr:    \[false, false, true \]\[TIER-1\],

    // Animation  
    useGPGPU:          \[false, false, true \]\[TIER-1\],  
    particleFPS:       \[20,    30,    60   \]\[TIER-1\],

    // Shader complexity  
    noiseOctaves:      \[2,     3,     5   \]\[TIER-1\],

FIX B — RENDERER SETUP

\`\`\`javascript  
  // Apply in your main Canvas/renderer setup

  import { SETTINGS } from './utils/deviceTier'

  // R3F Canvas:  
  \<Canvas  
    dpr={SETTINGS.pixelRatio}       // ← critical  
    gl={{  
      antialias: SETTINGS.TIER \>= 2,  
      powerPreference: 'high-performance',  
      stencil: false,               // never needed  
      depth: true,  
      alpha: false,                 // black bg \= no alpha

    frameloop="demand"              // ← MAJOR win:  
                                    // only render when  
                                    // something changes  
  \>  
\`\`\`

  IMPORTANT: Change frameloop from "always" to "demand"  
  and call invalidate() only when uniforms change.  
  This alone fixes 40% of the lag on static moments.

  For sections where constant rendering IS needed  
  (active scroll, particle animation):  
    Use frameloop="always" locally via the   
    useFrame hook returning early when not in viewport.
